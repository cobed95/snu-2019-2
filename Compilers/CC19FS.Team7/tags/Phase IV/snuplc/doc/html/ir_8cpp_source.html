609 void CCodeBlock::CleanupControlFlow(void)
610 {
611  list<CTacInstr*>::iterator it = _ops.begin();
612 
613  // 1. pass: delete all branches (absolute/conditional) that jump to the
614  // immediately next instruction. Deleting branch instruction will
615  // automatically decrease the reference count of the target label.
616  while (it != _ops.end()) {
617  CTacInstr *instr = *it++;
618 
619  if (instr->IsBranch()) {
620  CTacLabel *lbl = dynamic_cast<CTacLabel*>(instr->GetDest());
621  CTacInstr *next = (it == _ops.end() ? NULL : *it);
622 
623  if ((lbl != NULL) && (lbl == next)) {
624  delete instr;
625  it = _ops.erase(--it);
626  }
627  }
628  }
629 
630  // 2. pass: remove all labels with reference count 0
631  it = _ops.begin();
632  while (it != _ops.end()) {
633  CTacInstr *instr = *it++;
634 
635  CTacLabel *lbl = dynamic_cast<CTacLabel*>(instr);
636 
637  if ((lbl != NULL) && (lbl->GetRefCnt() == 0)) {
638  delete lbl;
639  it = _ops.erase(--it);
640  }
641  }
642 
643  // 3. renumber instructions (we shouldn't do that really, but it's prettier)
644  _inst_id = 0;
645  it = _ops.begin();
646  while (it != _ops.end()) (*it++)->SetId(_inst_id++);
647 }
