436 CAstStatement* CParser::statSequence(CAstScope *s)
437 {
438  //
439  // statSequence ::= [ statement { ";" statement } ].
440  // statement ::= assignment.
441  //
442  // FIRST(statSequence) = { tLetter }
443  // FOLLOW(statSequence) = { tElse, tEnd }
444  //
445  // FIRST(statement) = { tLetter }
446  // FOLLOW(statement) = { tSemicolon, tDot }
447  //
448 
449  // The linking of statement sequences is a bit akward here because
450  // we implement statSequence as a loop and not recursively.
451  // We keep a 'head' that points to the first statement and is finally
452  // returned at the end of the function. Head can be NULL if no statement
453  // is present.
454  // In the loop, we track the end of the linked list using 'tail' and
455  // attach new statements to that tail.
456  CAstStatement *head = NULL;
457 
458  if (_scanner->Peek().GetType() != tElse && _scanner->Peek().GetType() != tEnd) {
459  CAstStatement *tail = NULL;
460 
461  do {
462  CAstStatement *st = NULL;
463 
464  st = statement(s);
465 
466  assert(st != NULL);
467  if (head == NULL) head = st;
468  else tail->SetNext(st);
469  tail = st;
470 
471  EToken tt = _scanner->Peek().GetType();
472  if (tt == tEnd || tt == tElse) break;
473 
474  Consume(tSemicolon);
475  } while (!_abort);
476  }
477 
478  return head;
479 }
